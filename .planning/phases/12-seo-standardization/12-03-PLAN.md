1. Create `src/app/concepts/js/JSConceptsClient.tsx`:

2. Create new `src/app/concepts/js/page.tsx`:
```typescript
import type { Metadata } from 'next'
import { concepts } from '@/data/concepts'
import JSConceptsClient from './JSConceptsClient'
import { StructuredData } from '@/components/StructuredData'
import { generateBreadcrumbSchema } from '@/lib/seo/breadcrumb'

export const metadata: Metadata = {
  title: 'JavaScript Concepts - Interactive JS Learning | JS Interview Prep',
  description: `Master ${concepts.length} core JavaScript concepts: Closures, Event Loop, Prototypes, Promises, and more. Interactive visualizations and interview tips.`,
  keywords: 'javascript concepts, closures, event loop, prototypes, promises, hoisting, this keyword, javascript tutorial',
  openGraph: {
    title: 'JavaScript Concepts - Interactive Learning',
    description: 'Master core JavaScript concepts with interactive visualizations.',
    url: 'https://jsinterview.dev/concepts/js',
  },
  alternates: {
    canonical: '/concepts/js',
  },
}

export default function JSConceptsPage() {
  const breadcrumbSchema = generateBreadcrumbSchema([
    { name: 'Home', path: '/' },
    { name: 'Concepts', path: '/concepts' },
    { name: 'JavaScript' },
  ])

  return (
    <>
      <StructuredData data={breadcrumbSchema} />
      <JSConceptsClient />
    </>
  )
}
```

This pattern keeps client interactivity while enabling server-side metadata.
